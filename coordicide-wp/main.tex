\makeglossaries

\newglossaryentry{node}{name=Node, description={A machine which is part of the IOTA network. Its role is to issue new transactions and to validate already existing ones}}
\newglossaryentry{coo}{name=Coordinator, description={A trusted entity issuing milestones in order to guarantee finality and protect the Tangle against attacks}}
\newglossaryentry{milestone}{name=Milestone, description={A special transaction issued by the Coordinator. One of its features is to determine finality of the transactions it approves}}
\newglossaryentry{tx}{name=Transaction, description={A message that transfers funds or information between two nodes. A transaction is \textit{solid} if its entire history is known}}
\newglossaryentry{history}{name=History, description={The list of transactions directly or indirectly approved by a given transaction}}
\newglossaryentry{genesis}{name=Genesis, description={The first transaction ever generated in the Tangle}}
\newglossaryentry{sybil-attack}{name=Sybil attack, description={A Sybil attack is an attempt to gain control over a peer-to-peer network by forging multiple fake identities}}
\newglossaryentry{eclipse-attack}{name=Eclipse attack, description={A cyber-attack that aims to isolate and attack a specific user, rather than the whole network}}
\newglossaryentry{salt}{name=Salt, description={A random number used as an additional input to a one-way function that hashes data}}
\newglossaryentry{tip}{name=Tip, description={A transaction that has not been approved yet}}
\newglossaryentry{dictionary-attack}{name=Dictionary attack, description={A form of brute force attack technique for defeating an authentication mechanism by trying to determine its passphrase by trying millions of likely possibilities, such as words in a dictionary}}
\newglossaryentry{rainbow-table-attack}{name=Rainbow table attack, description={A precomputed table for reversing cryptographic hash functions}}

\newglossaryentry{consensus}{name=Consensus, description={The problem of agreeing on a specific data or value in distributed multi-agent systems in presence of faulty processes}}
\newglossaryentry{peering}{name=Peering, description={The procedure of discovering and connecting to other network nodes}}
\newglossaryentry{random-walk}{name=Random walk, description={A mathematical object that describes a path that consists of a succession of random steps on some mathematical space}}
\newglossaryentry{proof-of-work}{name=Proof-of-Work, description={A piece of data which is difficult (costly, time-consuming) to produce but easy for others to verify and which satisfies certain requirements}}
\newglossaryentry{neighbor}{name=Neighboring nodes, description={Nodes sharing the same link in a network}}
\newglossaryentry{byzantine-node}{name=Byzantine node, description={A participant in a distributed system which tries to damage its operation intentionally, e.g., by not forwarding messages to other participants and/or sending conflicting messages to different participants}}
\newglossaryentry{heartbeat}{name=Heartbeat, description={A periodic signal generated by hardware or software to indicate normal operation or to synchronize other parts of a computer system}}
\newglossaryentry{social-eng}{name=Social engineering, description={The use of deception to manipulate individuals into divulging confidential or personal information that may be used for fraudulent purposes}}
